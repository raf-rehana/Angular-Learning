## 4️⃣ Services

> 💡 **Service** = handles all API calls (GET, POST, PUT, DELETE). Components use services to get data.

**`department.service.ts`**
```typescript
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { environment } from '../../environments/environment';
import { DepartmentModel } from '../model/department.model';

@Injectable({ providedIn: 'root' })
export class DepartmentService {

  private url = environment.baseApi + 'department';

  constructor(private http: HttpClient) {}

  // GET all departments
  getAll(): Observable<DepartmentModel[]> {
    return this.http.get<DepartmentModel[]>(this.url);
  }

  // GET one department by id
  getById(id: string): Observable<DepartmentModel> {
    return this.http.get<DepartmentModel>(`${this.url}/${id}`);
  }

  // POST create new department
  save(dept: DepartmentModel): Observable<DepartmentModel> {
    return this.http.post<DepartmentModel>(this.url, dept);
  }

  // PUT update existing department
  update(dept: DepartmentModel): Observable<DepartmentModel> {
    return this.http.put<DepartmentModel>(`${this.url}/${dept.id}`, dept);
  }

  // DELETE remove department
  delete(id: string): Observable<void> {
    return this.http.delete<void>(`${this.url}/${id}`);
  }
}
```

### List Component Pattern (same for all 3)

**`list-student.ts`**
```typescript
import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { RouterModule } from '@angular/router';
import { StudentService } from '../../../services/student.service';
import { StudentModel } from '../../model/student.model';

@Component({
  selector: 'app-list-student',
  imports: [CommonModule, RouterModule],
  templateUrl: './list-student.html',
  styleUrl: './list-student.css'
})
export class ListStudent implements OnInit {

  studentList: StudentModel[] = [];  // stores list
  isLoading = false;
  errorMessage = '';

  constructor(private studentService: StudentService) {}

  // runs automatically when page loads
  ngOnInit(): void {
    this.loadAll();
  }

  loadAll(): void {
    this.isLoading = true;
    this.studentService.getAll().subscribe({
      next: (data) => {
        this.studentList = data;   // save data
        this.isLoading = false;
      },
      error: () => {
        this.errorMessage = 'Failed to load!';
        this.isLoading = false;
      }
    });
  }

  delete(id: string): void {
    if (confirm('Are you sure?')) {
      this.studentService.delete(id).subscribe({
        next: () => this.loadAll(),  // refresh list
        error: () => this.errorMessage = 'Delete failed!'
      });
    }
  }
}
```

### Add/Edit Component Pattern (same for all 3)

**`add-edit-student.ts`**
```typescript
import { Component, OnInit } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule } from '@angular/forms';
import { ActivatedRoute, Router, RouterModule } from '@angular/router';
import { StudentService } from '../../../services/student.service';
import { DepartmentService } from '../../../services/department.service';
import { StudentModel } from '../../model/student.model';
import { DepartmentModel } from '../../model/department.model';

@Component({
  selector: 'app-add-edit-student',
  imports: [CommonModule, FormsModule, RouterModule],
  templateUrl: './add-edit-student.html',
  styleUrl: './add-edit-student.css'
})
export class AddEditStudent implements OnInit {

  isEditMode = false;
  isLoading = false;
  errorMessage = '';
  successMessage = '';

  // department dropdown list
  departmentList: DepartmentModel[] = [];

  // empty student object
  student: StudentModel = {
    id: '',
    name: '',
    email: '',
    phone: '',
    fee: 0,
    departmentId: ''
  };

  constructor(
    private studentService: StudentService,
    private departmentService: DepartmentService,
    private route: ActivatedRoute,
    private router: Router
  ) {}

  ngOnInit(): void {
    this.loadDepartments();          // load dropdown
    const id = this.route.snapshot.paramMap.get('id');
    if (id) {
      this.isEditMode = true;
      this.loadStudent(id);          // load existing data
    }
  }

  loadDepartments(): void {
    this.departmentService.getAll().subscribe({
      next: (data) => this.departmentList = data
    });
  }

  loadStudent(id: string): void {
    this.isLoading = true;
    this.studentService.getById(id).subscribe({
      next: (data) => {
        this.student = data;
        this.isLoading = false;
      },
      error: () => {
        this.errorMessage = 'Failed to load student!';
        this.isLoading = false;
      }
    });
  }

  save(): void {
    this.isEditMode ? this.update() : this.create();
  }

  create(): void {
    this.isLoading = true;
    this.studentService.save(this.student).subscribe({
      next: () => this.router.navigate(['/student']),
      error: () => {
        this.errorMessage = 'Failed to create!';
        this.isLoading = false;
      }
    });
  }

  update(): void {
    this.isLoading = true;
    this.studentService.update(this.student).subscribe({
      next: () => this.router.navigate(['/student']),
      error: () => {
        this.errorMessage = 'Failed to update!';
        this.isLoading = false;
      }
    });
  }

  reset(): void {
    this.student = { id: '', name: '', email: '', phone: '', fee: 0, departmentId: '' };
    this.errorMessage = '';
    this.successMessage = '';
  }
}
```
